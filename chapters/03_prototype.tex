\section{Prototype Skeleton}

\subsection{Tech Stack}
\begin{itemize}
  \item \textbf{Backend:} Node.js + Express
  \item \textbf{Message Bus:} RabbitMQ
  \item \textbf{Relational DB:} PostgreSQL
  \item \textbf{Graph DB:} Neo4j
\end{itemize}
